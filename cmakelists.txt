	cmake_minimum_required(VERSION 2.8.9)
	project(neural_network)

	message(STATUS "Binary Directory for this make is: ${CMAKE_BINARY_DIR}")
	message(STATUS "Source Directory for this make is: ${CMAKE_SOURCE_DIR}")

###########################################################################################
# SUB DIRECTORIES
###########################################################################################

# Add Test Library
	add_subdirectory("${CMAKE_SOURCE_DIR}/test_library/")
	include_directories(${CMAKE_SOURCE_DIR}/test_library/)
	set(HEADER_SOURCES ${HEADER_SOURCES} ${CMAKE_SOURCE_DIR}/test_library/test_library.h)

###########################################################################################
# COMMMON
###########################################################################################
	include_directories("${CMAKE_SOURCE_DIR}/")
	set(HEADER_SOURCES ${HEADER_SOURCES} ${CMAKE_SOURCE_DIR}/NeuralNetwork.h)

###########################################################################################
# PROJECT
###########################################################################################
	set(SOURCES NeuralNetwork.c ${HEADER_SOURCES})
	set(EXECUTABLE_OUTPUT_PATH ${CMAKE_SOURCE_DIR}/bin)

	add_executable(neuralNetworkProject ${SOURCES})
	target_link_libraries(neuralNetworkProject test_library)

	install(TARGETS neuralNetworkProject DESTINATION ${EXECUTABLE_OUTPUT_PATH})
